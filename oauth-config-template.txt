# CONFIGURACIÓN OAUTH GITHUB - COPIA ESTOS VALORES A TU .env

# 1. Después de crear la GitHub App, reemplaza estos valores:
GITHUB_CLIENT_ID=tu_client_id_aqui
GITHUB_CLIENT_SECRET=tu_client_secret_aqui

# 2. Habilita la autenticación OAuth de GitHub:
NUXT_PUBLIC_USING_GITHUB_AUTH=true

# 3. Configuración de sesión (ya debería estar configurada):
SESSION_PASSWORD=hdi_super_secret_session_password_2025_32_chars

# 4. Configuración de la aplicación:
NUXT_PUBLIC_SCOPE=organization
NUXT_PUBLIC_GITHUB_ORG=hdicl
NUXT_PUBLIC_IS_DATA_MOCKED=true

# 5. Token de GitHub (opcional si usas OAuth):
GITHUB_TOKEN=ghp_your_github_token_here

# INSTRUCCIONES:
# 1. Crea la GitHub App en: https://github.com/organizations/hdicl/settings/apps
# 2. Copia el Client ID y Client Secret
# 3. Reemplaza los valores en tu archivo .env
# 4. Reinicia el servidor con: npm run dev
# 5. Ve a: http://localhost:3000
